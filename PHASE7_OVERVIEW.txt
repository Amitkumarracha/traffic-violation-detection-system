# ============================================================================
# THREAD ARCHITECTURE
# ============================================================================

"""
THREAD 1: CameraStream (from Phase 6)
┌─────────────────────────────────────────────────────────────────┐
│ Captures: 30 FPS from camera or video                          │
│ Output: capture_queue (maxsize=2)                              │
│ Non-blocking: Frame dropping if behind                          │
│ Speed: ~33ms per frame                                          │
└─────────────────────────────────────────────────────────────────┘
                              ↓
THREAD 2: PreprocessThread (_preprocess_thread)
┌─────────────────────────────────────────────────────────────────┐
│ 1. Read frame from capture_queue (non-blocking)               │
│ 2. Convert BGR → LAB color space                              │
│ 3. Apply CLAHE to L channel (contrast enhancement)           │
│ 4. Convert back to BGR                                         │
│ 5. Letterbox to 416×416 (maintains aspect ratio)             │
│ 6. Put FrameData in infer_queue                              │
│                                                                 │
│ Output: infer_queue (maxsize=2)                              │
│ Speed: ~5ms per frame                                          │
└─────────────────────────────────────────────────────────────────┘
                              ↓
THREAD 3: InferenceThread (_inference_thread)
┌─────────────────────────────────────────────────────────────────┐
│ 1. Read FrameData from infer_queue (non-blocking)            │
│ 2. Run YOLO26n detection (5-8ms)                             │
│ 3. Run DeepSort tracking (2-4ms)                             │
│ 4. Apply violation gate (4-stage filter, 1-2ms)             │
│ 5. Put InferenceResult in result_queue                       │
│                                                                 │
│ Output: result_queue (maxsize=4)                             │
│ Speed: ~10-15ms per frame                                     │
│ FPS Log: Every 100 frames                                     │
└─────────────────────────────────────────────────────────────────┘
                              ↓
THREAD 4: LogThread (_log_thread)
┌─────────────────────────────────────────────────────────────────┐
│ 1. Read InferenceResult from result_queue (non-blocking)     │
│ 2. For each violation:                                        │
│    a. Extract plate crop from frame                          │
│    b. Optional SRGAN: If plate <30px height, upscale 4×     │
│    c. Run PlateOCR.read_plate() on crop                     │
│    d. Get GPS coordinates from GPSReader                    │
│    e. Save to database via save_violation()                │
│    f. Push to cloud_queue                                   │
│    g. Log: "VIOLATION: type | Plate: text | GPS: lat,lon"  │
│                                                                 │
│ Output: cloud_queue (maxsize=10)                             │
│ Speed: 500-750ms per violation (async, non-blocking)        │
└─────────────────────────────────────────────────────────────────┘
                              ↓
MAIN THREAD: DisplayLoop (_display_loop)
┌─────────────────────────────────────────────────────────────────┐
│ 1. Get latest frame from result_queue (drain for latest)     │
│ 2. Draw overlay:                                              │
│    - FPS counter (top-left, green)                           │
│    - Violation count (top-right, red)                        │
│    - Queue depths (top-center, yellow)                       │
│    - Recent violations (bottom-left, cyan)                   │
│ 3. Display with cv2.imshow()                                │
│ 4. Check for 'q' key to quit                                │
│                                                                 │
│ Speed: ~33ms @ 30fps                                          │
│ Blocking: YES (main thread holds here)                       │
└─────────────────────────────────────────────────────────────────┘
"""

# KEY DATA STRUCTURES
# ============================================================================

"""
FrameData (Thread 1 → Thread 2)
───────────────────────────────
@dataclass
class FrameData:
    frame_id: int              # Sequential frame number
    timestamp: float           # Capture timestamp (seconds)
    frame: np.ndarray         # Enhanced frame (720×1280×3 BGR)
    processed: np.ndarray     # Letterboxed for inference (416×416×3)
    height: int               # Original height
    width: int                # Original width


InferenceResult (Thread 3 → Thread 4)
──────────────────────────────────────
@dataclass
class InferenceResult:
    frame_id: int             # From FrameData
    timestamp: float          # From FrameData
    frame: np.ndarray         # Original for display/OCR
    detections: list          # YOLO detections (all)
    tracks: dict              # DeepSort track_id → Track
    violations: list          # Confirmed by gate (filtered)


LogEntry (Thread 4 → Cloud)
──────────────────────────
@dataclass
class LogEntry:
    violation_id: str         # "frame_123_track_45"
    timestamp: float          # Violation time
    violation_type: str       # "without_helmet", "triple_ride"
    plate_text: str           # "MH12AB1234"
    confidence: float         # OCR confidence 0-1
    gps_coords: Tuple        # (latitude, longitude)
    frame_path: Optional[str] # Path to saved frame
    srgan_applied: bool       # Was upscaling used?
"""
